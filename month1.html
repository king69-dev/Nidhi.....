# Day 4: For Loops

# Basic for loop
for i in range(5):
    print(f"Count: {i}")

# Loop through a list
fruits = ['apple', 'banana', 'cherry']
for fruit in fruits:
    print(f"I like {fruit}")

# Multiplication table
number = 5
print(f"Multiplication table of {number}:")
for i in range(1, 11):
    print(f"{number} x {i} = {number * i}")

# Using break
print("\nUsing break:")
for i in range(10):
    if i == 5:
        break
    print(i)

# Using continue
print("\nUsing continue:")
for i in range(5):
    if i == 2:
        continue
    print(i)

# Nested loop - pattern
print("\nPattern:")
for i in range(5):
    for j in range(i + 1):
        print("*", end="")
    print()

# Day 5: While Loops

# Basic while loop
count = 0
while count < 5:
    print(f"Count is: {count}")
    count += 1

# Countdown
num = 10
print("\nCountdown:")
while num > 0:
    print(num)
    num -= 1
print("Blast off!")

# Sum until threshold
total = 0
number = 1
while total < 50:
    total += number
    number += 1
print(f"\nSum reached {total} with {number-1} numbers")

# Menu-driven program
print("\n=== Simple Menu ===")
choice = ''
while choice != '3':
    print("1. Say Hello")
    print("2. Say Goodbye")
    print("3. Exit")
    choice = input("Enter choice: ")
    
    if choice == '1':
        print("Hello!")
    elif choice == '2':
        print("Goodbye!")
    elif choice == '3':
        print("Exiting...")
    else:
        print("Invalid choice!")

# Day 7: Advanced Functions

# *args - variable number of arguments
def sum_all(*numbers):
    total = 0
    for num in numbers:
        total += num
    return total

print(sum_all(1, 2, 3))        # 6
print(sum_all(1, 2, 3, 4, 5))  # 15

# **kwargs - keyword arguments
def student_info(**info):
    for key, value in info.items():
        print(f"{key}: {value}")

student_info(name="John", age=20, grade="A", city="Mumbai")

# Lambda functions
square = lambda x: x ** 2
print(f"Square of 5: {square(5)}")

add = lambda a, b: a + b
print(f"3 + 4 = {add(3, 4)}")

# Lambda with map
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, numbers))
print(f"Squared: {squared}")

# Lambda with filter
even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
print(f"Even numbers: {even_numbers}")

# Scope example
global_var = "I'm global"

def test_scope():
    local_var = "I'm local"
    print(global_var)
    print(local_var)

test_scope()

# Week 1 Review - Calculator
def calculator():
    """Simple calculator with all operations"""
    def add(a, b): return a + b
    def subtract(a, b): return a - b
    def multiply(a, b): return a * b
    def divide(a, b): return a / b if b != 0 else "Error: Division by zero"
    
    return {
        'add': add,
        'subtract': subtract,
        'multiply': multiply,
        'divide': divide
    }

calc = calculator()
print(f"10 + 5 = {calc['add'](10, 5)}")
print(f"10 * 5 = {calc['multiply'](10, 5)}")
